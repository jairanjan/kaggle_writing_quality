import re
import math
def extract_nums(s):
    # Use regular expression to extract numbers inside square brackets
    numbers = re.findall(r'\d+', s)
    #print(s)
    #print(numbers)
    if numbers:
        x1=int(numbers[0])
        y1=int(numbers[1])
        x2=int(numbers[2])
        y2=int(numbers[3])
        return str(math.sqrt((x1-x2)**2+(y1-y2)**2))
    else:
        return "no_movement"
